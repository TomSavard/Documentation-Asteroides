36bool isCollide(Entite *a,Entite *b)
37
38{
39 return (b->getx() - a->getx())*(b->getx() - a->getx())+
40 (b->gety() - a->gety())*(b->gety() - a->gety())<
41 (a->getR() + b->getR())*(a->getR() + b->getR());
42}
